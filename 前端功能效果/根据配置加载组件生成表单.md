# vue项目根据配置加载组件生成表单

```js
  子组件里封装一个统一的事件派发方法，比如 emitFieldChange(model)。
  当某个字段值变化时，这个方法会拿到对应的 model 和当前值，然后通过
  $emit('field-change', { model, value }) 向父组件派发事件
  父组件只需要通过 @field-change="handleFieldChange" 来监听，就能集中处理所有字段变化逻辑
  
// formConfig.js

 <DynamicForm :config="formConfig" @field-change="handleFieldChange" @submit="handleSubmit" />
  handleFieldChange({ model, value }) {
    console.log(`字段 [${model}] 变化：`, value);
  },
  handleSubmit(formData) {
    console.log('提交表单：', formData);
    this.formData = formData;
  }
export const formConfig = [
  { type: 'input', label: '用户名', model: 'username', placeholder: '请输入用户名' },
  { type: 'password', label: '密码', model: 'password', placeholder: '请输入密码' },
  { type: 'select', label: '性别', model: 'gender', options: [
      { label: '男', value: 'male' },
      { label: '女', value: 'female' }
    ] 
  },
  { type: 'checkbox', label: '爱好', model: 'hobbies', options: [
      { label: '篮球', value: 'basketball' },
      { label: '足球', value: 'football' }
    ]
  }
];

```
```html
<template>
  <form @submit.prevent="handleSubmit" class="dynamic-form">
    <div
      v-for="(field, index) in config"
      :key="index"
      class="form-item"
    >
      <label class="form-label">{{ field.label }}：</label>

      <!-- input -->
      <input
        v-if="field.type === 'input'"
        v-model="formData[field.model]"
        :placeholder="field.placeholder"
        @input="emitFieldChange(field)"
      />

      <!-- password -->
      <input
        v-else-if="field.type === 'password'"
        type="password"
        v-model="formData[field.model]"
        :placeholder="field.placeholder"
        @input="emitFieldChange(field)"
      />

      <!-- select -->
      <select
        v-else-if="field.type === 'select'"
        v-model="formData[field.model]"
        @change="emitFieldChange(field)"
      >
        <option
          v-for="option in field.options"
          :key="option.value"
          :value="option.value"
        >
          {{ option.label }}
        </option>
      </select>

      <!-- checkbox -->
      <div v-else-if="field.type === 'checkbox'" class="checkbox-group">
        <label
          v-for="option in field.options"
          :key="option.value"
          class="checkbox-item"
        >
          <input
            type="checkbox"
            :value="option.value"
            v-model="formData[field.model]"
            @change="emitFieldChange(field)"
          />
          {{ option.label }}
        </label>
      </div>
    </div>

    <button type="submit" class="submit-btn">提交</button>
  </form>
</template>

<script>
export default {
  name: 'DynamicForm',
  props: {
    config: {
      type: Array,
      required: true
    }
  },
  data() {
    return {
      formData: this.initFormData(this.config)
    };
  },
  methods: {
    // 初始化表单数据
    initFormData(config) {
      const data = {};
      config.forEach(field => {
        data[field.model] = field.type === 'checkbox' ? [] : '';
      });
      return data;
    },

    // 触发字段变化事件
    emitFieldChange(field) {
      this.$emit('field-change', {
        model: field.model,
        value: this.formData[field.model]
      });
    },

    // 提交表单事件
    handleSubmit() {
      this.$emit('submit', { ...this.formData });
    }
  }
};
</script>
```
